/**

Componente de cargado de la página
@vue-data {number} numDots - Número de puntos en el efecto de carga
*/
export default {
  data() {
    return {
      numDots: 0
    }
  },
  created() {
    /**
     * Cambia el valor de la variable {numDots} de 0 a 3 en el lapso de 1.2s
     */
    setInterval(() => {
      if (this.numDots < 3) {
        this.numDots++
      } else {
        this.numDots = 0
      }
    }, 1250)
  },
  computed: {
    /**
     * Repite el punto tantas veces como diga la variable
     */
    loadingText() {
      return `${'.'.repeat(this.numDots)}`
    }
  }
}
